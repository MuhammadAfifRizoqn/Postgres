create table regions(
	region_id serial primary key,
	region_name varchar(25) 
);

create table countries(
	country_id varchar(2) primary key,
	country_name varchar(60),
	region_id integer not null,
	foreign key (region_id) references regions(region_id) on update cascade on delete cascade
);

create table locations(
	location_id varchar(2)primary key,
	street_address varchar(150),
	postal_code varchar(40),
	city varchar(60),
	state_province varchar (80),
	country_id varchar(2) not null,
	foreign key(country_id) references countries(country_id) on update cascade on delete cascade
);
alter table locations alter column location_id type varchar(5) // varchar di location kurang (2) dan dirubah menjadi (5)

create table departments(
	department_id varchar(2)primary key,
	department_name varchar(100),
	location_id varchar(2) not null,
	foreign key(location_id) references locations(location_id) on update cascade on delete cascade
);

alter table departments alter column location_id type varchar(5) // varchar di location kurang (2) dan dirubah menjadi (5)

create table employees(
	employee_id varchar(3)primary key,
	first_name varchar(60),
	last_name varchar(40),
	email varchar(60),
	phone_number varchar (60),
	hire_date DATE NOT NULL,
	job_id varchar(4),
	salary varchar (40),
	manager_id varchar (3),
	department_id varchar(2) not null,
	foreign key(manager_id) references employees(employee_id) on update cascade on delete cascade,
	foreign key(department_id) references departments(department_id) on update cascade on delete cascade
);


alter table employees add constraint job_id foreign key(job_id) references jobs(job_id) on update cascade on delete cascade // saya lupa untuk job_idnya di pembuatan table employees

create table dependents(
	dependent_id varchar(3)primary key,
	first_name varchar(60),
	last_name varchar(40),
	relationship varchar (100),
	employee_id varchar (2) not NULL,
	foreign key(employee_id) references employees(employee_id) on update cascade on delete cascade
);

alter table dependents alter column employee_id type varchar(5) // varchar di employee_id kurang (2) dan dirubah menjadi (5)
